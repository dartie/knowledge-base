```javascript
/* listener for showing icons during collapse/expand */

$(".panel-collapse").on("hide.bs.collapse", function () {
    parent_element = $(this).parent();  // get the parent element which has the text which needs to be modified
    current_text = parent_element.find(".collapsepanelheader").text();
    //parent_element.find(".collapsepanelheader").html(current_text + ' <span class="glyphicon glyphicon-collapse-down"></span>');
    parent_element.find(".collapsepanelheader").html(current_text + ' <span class="glyphicon glyphicon glyphicon-plus-sign pull-left"> </span>');
});
$(".panel-collapse").on("show.bs.collapse", function () {
    parent_element = $(this).parent();  // get the parent element which has the text which needs to be modified
    current_text = parent_element.find(".collapsepanelheader").text();
    //parent_element.find(".collapsepanelheader").html(current_text + ' <span class="glyphicon glyphicon-collapse-up"></span>');
    parent_element.find(".collapsepanelheader").html(current_text + ' <span class="glyphicon glyphicon glyphicon-minus-sign pull-left"> </span>');
});

```
